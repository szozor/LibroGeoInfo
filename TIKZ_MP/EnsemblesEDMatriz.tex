%

-- cycle;

\draw[domain=0:360,samples=200,thick] plot ({\ap*cos(\x)+\cpi},{\bp*sin(\x)});
\draw[dashed,domain=0:360,samples=200,thick] plot ({\ap*cos(\x)+\cpd},{\bp*sin(\x)});

\draw (.2,.45) node[right,scale=.75]{$\ED_\vcol$};
